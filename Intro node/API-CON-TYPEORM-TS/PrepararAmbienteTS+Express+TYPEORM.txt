1. Instalación inicial
npm init -y
npm install express cors morgan
npm install -D typescript ts-node @types/node @types/express @types/cors @types/morgan
npm install --save-dev @types/node
npm install -D nodemon
npm install reflect-metadata
npm install mysql2

2. Crear tsconfig.json
npx tsc --init

Luego ajusta con estas opciones clave:
"compilerOptions": {
    "esModuleInterop": true,
    "moduleResolution": "node",
    "emitDecoratorMetadata": true,
    "experimentalDecorators": true,

    "rootDir": "./src",
    "outDir": "./dist",

    "module": "CommonJS",
    "target": "ES2020",
    "types": [],

    "sourceMap": true,
    "declaration": true,
    "declarationMap": true,

    "noUncheckedIndexedAccess": true,
    "exactOptionalPropertyTypes": true,

    "strict": true,
    "jsx": "react-jsx",
    "verbatimModuleSyntax": false,
    "isolatedModules": true,
    "noUncheckedSideEffectImports": true,
    "moduleDetection": "force",
    "skipLibCheck": true,
  }

3. Ajustar package.json
{
  "name": "tu-proyecto",
  "version": "1.0.0",
  "scripts": {
    "dev": "ts-node src/index.ts",
    "build": "tsc",
    "start": "node dist/index.js"
  },
  "devDependencies": {
    "@types/express": "^5.0.0",
    "@types/node": "^20.0.0",
    "typescript": "^5.0.0",
    "ts-node": "^10.0.0"
  },
  "dependencies": {
    "express": "^5.0.0"
  }
}


4. Estructura de carpetas
proyecto/
├── src/
│   └── index.ts
├── dist/          (generado al compilar)
├── package.json
├── tsconfig.json
└── .gitignore

5. Comandos principales
npm run dev      # ejecuta en desarrollo (con hot reload si usas nodemon)
npm run build    # compila a JavaScript
npm start        # ejecuta el JavaScript compilado


instalar JWT
npm install jsonwebtoken bcrypt
npm install -D @types/jsonwebtoken @types/bcrypt
npm i --save-dev @types/jsonwebtoken

instalar validadores
npm install class-validator class-transformer

instalar bcrypt
npm i --save-dev @types/bcrypt
